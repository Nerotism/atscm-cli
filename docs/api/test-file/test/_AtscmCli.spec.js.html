    import expect from 'unexpected';
import { stub } from 'sinon';

import AtscmCli, { UsageError } from '../src/_AtscmCli';
import log from '../src/lib/util/Logger';

/** @test {AtscmCli} */
describe.skip('AtscmCli', function() {
  /** @test {AtScmCli#parseArguments} */
  describe('#parseArguments', function() {
    it('should fail with unknown options', function() {
      return expect((new AtscmCli(['--unknown'])).parseArguments(),
        'to be rejected with', new UsageError('Unknown argument: unknown')
      );
    });

    it('should store provided options', function() {
      const cli = new AtscmCli(['--silent']);

      return cli.parseArguments()
        .then(() => {
          expect(cli.options, 'to be defined');
          expect(cli.options.silent, 'to be true');
        });
    });

    it('should store provided command', function() {
      const cli = new AtscmCli([`${AtscmCli.Command.Docs}`]);

      return cli.parseArguments()
        .then(() => {
          expect(cli.command, 'to be defined');
          expect(cli.command, 'to equal', AtscmCli.Command.Docs);
        });
    });
  });
});


/** @test {AtscmCli} */
describe.skip('AtscmCli', function() {
  /** @test {AtscmCli#constructor} */
  describe('#constructor', function() {
    it('should store command line arguments', function() {
      const args = ['test', '--args'];
      const cli = new AtscmCli(args);

      expect(cli._argv, 'to equal', args);
    });
  });

  function launchCli(args, callback) {
    const cli = new AtscmCli(args);
    cli.launch(() => callback(cli));
  }

  function launchCliWithLocalModule(args, callback) {
    launchCli(args.concat(['--cwd', 'test/fixtures']), callback);
  }

  const logStub = stub(log, 'info');
  const errorStub = stub(log, 'error');

  after(function() {
    logStub.restore();
    errorStub.restore();
  });

  afterEach(function() {
    logStub.reset();
    errorStub.reset();
  });

  /** @test {AtscmCli#requireModule} */
  describe('#requireModule', function() {
    it('should return false if no local module is found', function() {
      launchCli([], cli => {
        expect(cli.requireModule(), 'to equal', false);
      });
    });

    it('should print an error if no local module is found', function() {
      launchCli([], () => {
        expect(errorStub.lastCall.args[0], 'to contain', `Local ${AtscmCli.BinName} not found`);
      });
    });

    it('should suggest install if no local module is found', function() {
      launchCli([], () => {
        expect(logStub.lastCall.args[0], 'to contain', `install ${AtscmCli.BinName}`);
      });
    });

    it('should return true if local module is found', function() {
      launchCliWithLocalModule([], cli => {
        expect(cli.requireModule(), 'to equal', false);
      });
    });
  });

  /** @test {AtscmCli#requireConfig} */
  describe('#requireConfig', function() {
    it('should return false if no config is found', function() {
      launchCli([], cli => {
        expect(cli.requireConfig(), 'to equal', false);
      });
    });

    it('should print an error if no config is found', function() {
      launchCli([], () => {
        expect(errorStub.lastCall.args[0], 'to contain', `No ${AtscmCli.ConfigName} found`);
      });
    });

    it('should return true if config is found', function() {
      launchCliWithLocalModule([], cli => {
        expect(cli.requireConfig(), 'to equal', false);
      });
    });
  });

  /** @test {AtscmCli#requireModuleAndConfig} */
  describe('#requireModuleAndConfig', function() {
    it('should return false if not run inside a project dir', function() {
      launchCli([], cli => {
        expect(cli.requireModuleAndConfig(), 'to equal', false);
      });
    });

    it('should return true if run inside project dir', function() {
      launchCliWithLocalModule([], cli => {
        expect(cli.requireModuleAndConfig(), 'to equal', false);
      });
    });
  });

  /** @test {AtscmCli#printVersion} */
  describe('#printVersion', function() {
    it('should print cli version if no local module is found', function(done) {
      launchCli(['--version'], cli => {
        cli.printVersion();

        expect(logStub.lastCall.args[0], 'to match', /CLI version/i);

        done();
      });
    });

    it('should print local version if local module is found', function(done) {
      launchCliWithLocalModule(['--version'], cli => {
        cli.printVersion();

        expect(logStub.lastCall.args[0], 'to match', /Local version/i);

        done();
      });
    });
  });

  /** @test {AtscmCli#printConfig} */
  describe('#printConfig', function() {
    it('should exit if module or config cannot be found', function(done) {
      launchCli(['--config'], cli => {
        cli.printConfig();

        expect(errorStub.lastCall.args[0], 'to match', /Local /);
        done();
      });
    });

    it('should print config if not module or config is found', function(done) {
      launchCliWithLocalModule(['--config'], cli => {
        cli.printConfig();

        expect(errorStub.lastCall.args[0], 'to match', /config /);

        done();
      });
    });
  });
