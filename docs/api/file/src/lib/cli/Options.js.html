    export class Option {

  constructor(desc, options = {}) {
    this.desc = desc;

    Object.keys(options)
      .forEach(k => (this[k] = options[k]));
  }

  static boolean(desc, options = {}) {
    return new this(desc, Object.assign(options, { type: 'boolean', default: false }));
  }

  static string(desc, options = {}) {
    return new this(desc, Object.assign(options, { type: 'string', requiresArg: true }));
  }

}

const Options = {
  cli: Option.boolean('Open CLI documentation.'),
  config: Option.boolean('Print project configuration. Useful for debugging.'),
  cwd: Option.string('Manually set the CWD.'),
  help: Option.boolean('Show this help'),
  logLevel: new Option('Set the Logger level. ' +
    '-L for least verbose and -LLLL for most verbose. -LLL is default.', {
      alias: 'L',
      count: true,
    }
  ),
  projectfile: Option.string('Manually set path of Atviseproject file to use. ' +
    'This will set the CWD to the Atviseproject file\'s directory as well.',
    { alias: 'p' }),
  silent: Option.boolean('Supress all logging.', { alias: 'S' }),
  version: Option.boolean('Print version.', { alias: 'v' }),
};

export default Options;

/**
 * Options that can be used globally
 * @type {{projectfile: *, cwd: *, config: *, version: *, help: *, silent: *, logLevel: Option}}
 */
export const global = {
  projectfile: Options.projectfile,
  cwd: Options.cwd,
  config: Options.config,
  version: Options.version,
  help: Options.help,
  silent: Options.silent,
  logLevel: Options.logLevel,
};

};

/**
 * Options that can be used with the "docs" command
 * @type {Object}
 */
export const docs = {
  cli: Options.cli,
};
